Add full interactive world population map with all countries function getColor(pop) { if (!pop || pop === "Population data not available") return "#ccc"; // grey if missing var n = pop.replace(/[^0-9]/g, ''); // remove letters n = parseInt(n); if (pop.includes("billion")) n *= 1000; // convert billion -> million if (n > 500) return "#4B0082"; // deep purple else if (n > 200) return "#6A0DAD"; // purple else if (n > 100) return "#2E8B57"; // dark green else if (n > 50) return "#3CB371"; // medium green else if (n > 20) return "#20B2AA"; // teal else if (n > 10) return "#00CED1"; // cyan else if (n > 1) return "#87CEFA"; // light blue else return "#E0FFFF"; // very light blue }
